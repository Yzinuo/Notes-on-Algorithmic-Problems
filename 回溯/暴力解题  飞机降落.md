#  暴力解法

因为数据量小，我们使用暴力解题。 dfs便利所有结果。

##   代码：
```
#include<bits/stdc++.h>
using namespace std;
struct plane{
    int a,w,g;
}p[50];
bool status[50];
int n,m;
bool dfs(int u,int time){
    if(u >= m)
        return 1;
    
    for(int i = 0; i < m; i++){
        if(!status[i]){
            status[i] = 1;
            if(time > p[i].a + p[i].w){
                status[i] = 0;
                continue;
            }
            int t = max(time,p[i].a) + p[i].g;
            if(dfs(u+1,t))
                return 1;
            
            status[i] = 0;
        }
    }
    return 0;
}
int main(){
    
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> n;
    while(n--){
        cin >> m;
        for(int i = 0; i < m; i++)
        {
            cin >> p[i].a>>p[i].w>>p[i].g;
        }
        memset(status,0,sizeof status);
        if(dfs(0,0))
            cout << "YES" << "\n";
        else
            cout << "NO" << "\n";
        
    }
    return 0;
}
```
